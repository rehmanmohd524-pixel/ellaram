pipeline {
    agent any

    // Define the tools you want to use
    tools {
        maven 'M3'      // Replace 'M3' with the exact name of your Maven installation in Jenkins
        jdk 'JDK11'     // Replace 'JDK11' with the exact name of your JDK installation in Jenkins
    }

    stages {
        stage('Checkout') {
            steps {
                // Checkout from your Git repository
                git url: 'https://github.com/rehmanmohd524-pixel/ellaram.git', branch: 'main'
            }
        }

        stage('Build') {
            steps {
                // Clean and package the project using Maven
                sh 'mvn clean package'
            }
        }

        stage('Test') {
            steps {
                // Run tests (JUnit)
                sh 'mvn test'
            }
        }

        stage('Archive') {
            steps {
                // Archive the WAR file for deployment
                archiveArtifacts artifacts: 'target/*.war', fingerprint: true
            }
        }
    }

    post {
        success {
            echo 'Build completed successfully!'
        }
        failure {
            echo 'Build failed. Check the logs!'
        }
    }
}
