pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Build') {
            steps {
                sh 'mvn clean compile'
            }
        }

        stage('Test') {
            steps {
                sh 'mvn test'
            }
        }

        stage('Package') {
            steps {
                sh 'mvn package'
            }
        }
    }

stage('deploy') { 
            steps {
                sh "cp /root/.m2/repository/DevOps/TestProject/1.0/TestProject-1.0.war /var/lib/tomcat10/webapps"
            }
        }
    }
}

    post {
        success {
            echo 'Build Successful!'
            archiveArtifacts artifacts: 'target/*.jar'
        }
        failure {
            echo 'Build Failed!'
        }
    }
}
